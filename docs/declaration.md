# Declaration
- List assignment
- Type inference

```
// x is inferred to be a number
let x = 7

// array
let array = []

// infers number array
arrayOrTree += 5

// must be a tree
arrayOrTree += "Hi"

// datatypes can be declared
bool x = true

// assign all variables seperately (number, string, bool, number array)
let a, b, c, d = 5, "good", true, [2, 3, 4]

// force all to be the same type
string firstname, middlename, lastname = "John", "Michael", "Smith"

// assign all variables to the same value
let initialValue, zero, lowest = 0
```
